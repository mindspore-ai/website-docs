46 ///\brief MindSpore predict session.
47 ///
48 /// This class represents session of MindSpore predict.
49 ///
50 ///\note
51 /// The caller needs to allocate and free memory of inputs and outputs.
52 /// New Session is not suggested, please use CreateSession function to create new session class.
53 class MSPREDICT_API Session {
54  public:
55  ///\brief Constructor of MindSpore predict session.
56  ///
57  ///\param[in] ctx The context of the session.
58  ///
59  ///\return Instance of MindSpore predict session.
60  explicit Session(const Context &ctx);
61 
62  ///\brief Destructor of MindSpore predict session.
63  ~Session();
64 
65  ///\brief Init the session.
66  ///
67  ///\param[in] ctx The context of the session.
68  ///\param[in] size The size of the session.
69  ///\param[in] graphBuf The buffer of the graph, used for build session.
70  ///
71  ///\return Return RET_OK if the initialization is success, otherwhise return RET_ERROR.
72  int Init(const char *graphBuf, size_t size);
73 
74  ///\brief Get the input of session.
75  ///
76  ///\return Input node's input tensors if found, empty vector otherwise.
77  ///
78  ///\note
79  /// The caller needs to allocate and free memory of inputs.
80  std::vector<Tensor *> GetInput();
81 
82  ///\brief Run the session.
83  ///
84  ///\param[in] inputs The input of the session.
85  ///
86  ///\return Return RET_OK if run success, otherwhise return RET_ERROR.
87  ///\note
88  /// Currently input tensors' data format only support FORMAT_NCHW.
89  /// Currently input tensors' data type only support FLOAT.
90  int Run(const std::vector<Tensor *> &inputs);
91 
92  ///\brief Get the output of session.
93  ///
94  ///\param[in] nodeName Given output node name.
95  ///
96  ///\return Output node's output tensors if found, empty vector otherwise.
97  ///
98  ///\note
99  /// The caller needs to free memory of outputs.
100  std::vector<Tensor *> GetOutput(const std::string &nodeName);
101 
102  ///\brief Get the all output of session.
103  ///
104  ///\return Every output node's output tensors.
105  ///
106  ///\note
107  /// The caller needs to free memory of outputs.
108  std::map<std::string, std::vector<Tensor *>> GetAllOutput();
109 
110  protected:
111  ///\brief Init the executor.
112  ///
113  ///\return Return RET_OK if the initialization is success, otherwhise return RET_ERROR.
114  int InitExecutor();
115 
116  const Context &_ctx;
117  Graph *_graph = nullptr;
118  GraphExecution *_executor = nullptr;
119  bool reinitExecutor = true;
120 };
