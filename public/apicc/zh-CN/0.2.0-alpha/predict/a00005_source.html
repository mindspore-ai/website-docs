24 /* Success */
25 constexpr int RET_OK = 0; /**< No error occurs. */

42 constexpr int RET_NOT_FIND_OP = -301; /**< Failed to find OP. */

44 constexpr int RET_INVALID_OP_ATTR = -303; /**< Invalid OP attr. */

48 constexpr int RET_FORMAT_ERR = -401; /**< Failed to checking tensor format. */
