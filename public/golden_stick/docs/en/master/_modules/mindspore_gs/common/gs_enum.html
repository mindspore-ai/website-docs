    supported quant cell type enums

    LINEAR = 'linear'
    CONV2D = 'conv2d'
    MF_LINEAR = 'mf_linear'

    uint4 = 'UINT4'
    int8 = 'INT8'
    uint8 = 'UINT8'
    int16 = 'INT16'
    UINT16 = 'UINT16'
