17 #ifndef MINDSPORE_LITE_INCLUDE_MS_TENSOR_H_
18 #define MINDSPORE_LITE_INCLUDE_MS_TENSOR_H_
19 
20 #include <utility>
21 #include <vector>
22 #include <memory>
23 #include "ir/dtype/type_id.h"
24 
25 namespace mindspore {
26 #define MS_API __attribute__((visibility("default")))
27 namespace tensor {
28 /// \brief MSTensor defined tensor in MindSpore Lite.
29 class MS_API MSTensor {
30  public:
31  /// \brief Constructor of MindSpore Lite MSTensor.
32  ///
33  /// \return Instance of MindSpore Lite MSTensor.
34  MSTensor() = default;
35 
36  /// \brief Static method to create a MSTensor pointer.
37  ///
38  /// \param[in] data_type Define data type of tensor to be created.
39  /// \param[in] shape Define Shape of tensor to be created.
40  ///
41  /// \note TypeId is defined in mindspore/mindspore/core/ir/dtype/type_id.h. Only number types in TypeId enum are
42  /// suitable for MSTensor.
43  ///
44  /// \return the pointer of MSTensor.
45  static MSTensor *CreateTensor(TypeId data_type, const std::vector<int> &shape);
46 
47  /// \brief Destructor of MindSpore Lite Model.
48  virtual ~MSTensor() = default;
49 
50  /// \brief Get data type of the MindSpore Lite MSTensor.
51  ///
52  /// \note TypeId is defined in mindspore/mindspore/core/ir/dtype/type_id.h. Only number types in TypeId enum are
53  /// suitable for MSTensor.
54  ///
55  /// \return MindSpore Lite TypeId of the MindSpore Lite MSTensor.
56  virtual TypeId data_type() const = 0;
57 
58  /// \brief Set data type for the MindSpore Lite MSTensor.
59  ///
60  /// \param[in] data_type Define MindSpore Lite TypeId to be set in the MindSpore Lite MSTensor.
61  ///
62  /// \return MindSpore Lite TypeId of the MindSpore Lite MSTensor after set.
63  virtual TypeId set_data_type(TypeId data_type) = 0;
64 
65  /// \brief Get shape of the MindSpore Lite MSTensor.
66  ///
67  /// \return A vector of int as the shape of the MindSpore Lite MSTensor.
68  virtual std::vector<int> shape() const = 0;
69 
70  /// \brief Set shape for the MindSpore Lite MSTensor.
71  ///
72  /// \param[in] shape Define a vector of int as shape to be set into the MindSpore Lite MSTensor.
73  ///
74  /// \return size of shape of the MindSpore Lite MSTensor after set.
75  virtual size_t set_shape(const std::vector<int> &shape) = 0;
76 
77  /// \brief Get size of the dimension of the MindSpore Lite MSTensor index by the parameter index.
78  ///
79  /// \param[in] index Define index of dimension returned.
80  ///
81  /// \return Size of dimension of the MindSpore Lite MSTensor.
82  virtual int DimensionSize(size_t index) const = 0;
83 
84  /// \brief Get number of element in MSTensor.
85  ///
86  /// \return Number of element in MSTensor.
87  virtual int ElementsNum() const = 0;
88 
89  /// \brief Get hash of the MindSpore Lite MSTensor.
90  ///
91  /// \return Hash of the MindSpore Lite MSTensor.
92  virtual std::size_t hash() const = 0;
93 
94  /// \brief Get byte size of data in MSTensor.
95  ///
96  /// \return Byte size of data in MSTensor.
97  virtual size_t Size() const = 0;
98 
99  /// \brief Get the pointer of data in MSTensor.
100  ///
101  /// \note The data pointer can be used to both write and read data in MSTensor.
102  ///
103  /// \return the pointer points to data in MSTensor.
104  virtual void *MutableData() const = 0;
105 };
106 } // namespace tensor
107 } // namespace mindspore
108 #endif // MINDSPORE_LITE_INCLUDE_MS_TENSOR_H_
