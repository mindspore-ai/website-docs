18 #define MINDSPORE_LITE_INCLUDE_LITE_SESSION_H
19 
20 #include <memory>
21 #include <vector>
22 #include <string>
23 #include <unordered_map>
24 #include "include/ms_tensor.h"
25 #include "include/model.h"
26 #include "include/context.h"

28 namespace mindspore {
29 namespace session {
30 /// \brief CallBackParam defined input arguments for callBack function.
31 struct CallBackParam {
32  std::string name_callback_param; /**< node name argument */
33  std::string type_callback_param; /**< node type argument */
34 };
35 
36 /// \brief KernelCallBack defined the function pointer for callBack.
37 using KernelCallBack = std::function<bool(std::vector<tensor::MSTensor *> inputs,
38  std::vector<tensor::MSTensor *> outputs, const CallBackParam &opInfo)>;
39 
40 /// \brief LiteSession defined session in MindSpore Lite for compiling Model and forwarding model.
41 class MS_API LiteSession {
42  public:
43  /// \brief Static method to create a LiteSession pointer.
44  ///
45  /// \param[in] context Define the context of session to be created.
46  ///
47  /// \return Pointer of MindSpore Lite LiteSession.
48  static LiteSession *CreateSession(lite::Context *context);
49 
50  /// \brief Destructor of MindSpore Lite LiteSession.
51  virtual ~LiteSession() = default;
52 
53  /// \brief Attempt to bind or unbind threads in the thread pool to or from the specified cpu core.
54  ///
55  /// \param[in] if_bind Define whether to bind or unbind threads.
56  virtual void BindThread(bool if_bind) = 0;
57 
58  /// \brief Compile MindSpore Lite model.
59  ///
60  /// \note CompileGraph should be called before RunGraph.
61  ///
62  /// \param[in] model Define the model to be compiled.
63  ///
64  /// \return STATUS as an error code of compiling graph, STATUS is defined in errorcode.h.
65  virtual int CompileGraph(lite::Model *model) = 0;
66 
67  /// \brief Get input MindSpore Lite MSTensors of model.
68  ///
69  /// \return The vector of MindSpore Lite MSTensor.
70  virtual std::vector<tensor::MSTensor *> GetInputs() const = 0;
71 
72  /// \brief Get input MindSpore Lite MSTensors of model by node name.
73  ///
74  /// \param[in] node_name Define node name.
75  ///
76  /// \return The vector of MindSpore Lite MSTensor.
77  virtual std::vector<tensor::MSTensor *> GetInputsByName(const std::string &node_name) const = 0;
78 
79  /// \brief Run session with callback.
80  ///
81  /// \param[in] before Define a call_back_function to be called before running each node.
82  /// \param[in] after Define a call_back_function called after running each node.
83  ///
84  /// \note RunGraph should be called after CompileGraph.
85  ///
86  /// \return STATUS as an error code of running graph, STATUS is defined in errorcode.h.
87  virtual int RunGraph(const KernelCallBack &before = nullptr, const KernelCallBack &after = nullptr) = 0;
88 
89  /// \brief Get output MindSpore Lite MSTensors of model mapped by node name.
90  ///
91  /// \return The map of output node name and MindSpore Lite MSTensor.
92  virtual std::unordered_map<std::string, std::vector<mindspore::tensor::MSTensor *>> GetOutputMapByNode() const = 0;
93 
94  /// \brief Get output MindSpore Lite MSTensors of model by node name.
95  ///
96  /// \param[in] node_name Define node name.
97  ///
98  /// \return The vector of MindSpore Lite MSTensor.
99  virtual std::vector<tensor::MSTensor *> GetOutputsByNodeName(const std::string &node_name) const = 0;
100 
101  /// \brief Get output MindSpore Lite MSTensors of model mapped by tensor name.
102  ///
103  /// \return The map of output tensor name and MindSpore Lite MSTensor.
104  virtual std::unordered_map<std::string, mindspore::tensor::MSTensor *> GetOutputMapByTensor() const = 0;
105 
106  /// \brief Get name of output tensors of model compiled by this session.
107  ///
108  /// \return The vector of string as output tensor names in order.
109  virtual std::vector<std::string> GetOutputTensorNames() const = 0;
110 
111  /// \brief Get output MindSpore Lite MSTensors of model by tensor name.
112  ///
113  /// \param[in] tensor_name Define tensor name.
114  ///
115  /// \return Pointer of MindSpore Lite MSTensor.
116  virtual mindspore::tensor::MSTensor *GetOutputByTensorName(const std::string &tensor_name) const = 0;
117 
118  /// \brief Resize inputs shape.
119  ///
120  /// \param[in] inputs Define the new inputs shape.
121  ///
122  /// \return STATUS as an error code of resize inputs, STATUS is defined in errorcode.h.
123  virtual int Resize(const std::vector<tensor::MSTensor *> &inputs) = 0;
124 };
125 } // namespace session
126 } // namespace mindspore
127 #endif // MINDSPORE_LITE_INCLUDE_LITE_SESSION_H
