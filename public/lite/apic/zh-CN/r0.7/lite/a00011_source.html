39 /// \brief Model defined model in MindSpore Lite for managing graph.
40 class MS_API Model {
41  public:
42  /// \brief Static method to create a Model pointer.
43  ///
44  /// \param[in] model_buf Define the buffer read from a model file.
45  /// \param[in] size Define bytes number of model buffer.
46  ///
47  /// \return Pointer of MindSpore Lite Model.
48  static Model *Import(const char *model_buf, size_t size);
49 
50  /// \brief Constructor of MindSpore Lite Model using default value for parameters.
51  ///
52  /// \return Instance of MindSpore Lite Model.
53  Model() = default;
54 
55  /// \brief Destructor of MindSpore Lite Model.
56  virtual ~Model();
57 
58  /// \brief Get MindSpore Lite Primitive by name.
59  ///
60  /// \param[in] name Define name of primitive to be returned.
61  ///
62  /// \return the pointer of MindSpore Lite Primitive.
63  PrimitiveC *GetOp(const std::string &name) const;
64 
65  /// \brief Get graph defined in flatbuffers.
66  ///
67  /// \return the pointer of graph defined in flatbuffers.
68  const schema::MetaGraph *GetMetaGraph() const;
69 
70  /// \brief Free MetaGraph in MindSpore Lite Model.
71  void FreeMetaGraph();
72 
73  protected:
74  ModelImpl *model_impl_ = nullptr;
75 };

77 /// \brief ModelBuilder defined by MindSpore Lite.
78 class MS_API ModelBuilder {
79  public:
80  /// \brief OutEdge defined by MindSpore Lite.
81  struct OutEdge {
82  std::string nodeId; /**< ID of a node linked by this edge */
83  size_t outEdgeIndex; /**< Index of this edge */
84  };
85 
86  /// \brief Constructor of MindSpore Lite Model using default value for parameters.
87  ///
88  /// \return Instance of MindSpore Lite ModelBuilder.
89  ModelBuilder() = default;
90 
91  /// \brief Destructor of MindSpore Lite ModelBuilder.
92  virtual ~ModelBuilder() = default;
93 
94  /// \brief Add primitive into model builder for model building.
95  ///
96  /// \param[in] op Define the primitive to be added.
97  /// \param[in] inputs Define input edge of primitive to be added.
98  ///
99  /// \return ID of the added primitive.
100  virtual std::string AddOp(const PrimitiveC &op, const std::vector<OutEdge> &inputs) = 0;
101 
102  /// \brief Finish constructing the model.
103  ///
104  /// \return the pointer of MindSpore Lite Model.
105  virtual Model *Construct();
106 };
