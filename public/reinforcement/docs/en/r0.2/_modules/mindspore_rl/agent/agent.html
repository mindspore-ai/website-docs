        >>> from mindspore_rl.agent.actor import Actor

        >>> actors = Actor()
        >>> learner = Learner()
        >>> agent = Agent(actors, learner)
        >>> print(agent)
        Agent<
        (_actors): Actor<>
        (_learner): Learner<>
        >
    """

    def __init__(self, actors, learner):
        super(Agent, self).__init__(auto_prefix=False)
        self._actors = actors
        self._learner = learner
